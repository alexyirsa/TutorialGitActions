413 

414 Set ``unique`` to ``False`` for random sampling with replacement, and set ``unique`` to

416

431 fake.random_elements(

441 fake.random_elements( 

442 elements=OrderedDict([ 

443 ("variable_1", 0.5), 

444 ("variable_2", 0.2), 

445 ("variable_3", 0.2), 

446 ("variable_4": 0.1),

448 )

453 :sample: elements=('a', 'b', 'c', 'd'), length=4, unique=True 

454 :sample: elements=OrderedDict([

456 ("b", 0.35),

460 :sample: elements=OrderedDict([ 

461 ("a", 0.45), 

462 ("b", 0.35),

555 def randomize_nb_elements( 

556 self, 

557 number: int = 10, 

558 le: bool = False, 

559 ge: bool = False, 

560 min: Optional[int] = None, 

561 max: Optional[int] = None, 

562 ) -> int: 

563 """Generate a random integer near ``number`` according to the following rules: 

564 

565 - If ``le`` is ``False`` (default), allow generation up to 140% of ``number``. 

566 If ``True``, upper bound generation is capped at 100%. 

567 - If ``ge`` is ``False`` (default), allow generation down to 60% of ``number``. 

568 If ``True``, lower bound generation is capped at 100%. 

569 - If a numerical value for ``min`` is provided, generated values less than ``min`` 

570 will be clamped at ``min``. 

571 - If a numerical value for ``max`` is provided, generated values greater than 

572 ``max`` will be clamped at ``max``. 

573 - If both ``le`` and ``ge`` are ``True``, the value of ``number`` will automatically 

574 be returned, regardless of the values supplied for ``min`` and ``max``. 

575 

576 :sample: number=100 

577 :sample: number=100, ge=True 

578 :sample: number=100, ge=True, min=120 

579 :sample: number=100, le=True 

580 :sample: number=100, le=True, max=80 

581 :sample: number=79, le=True, ge=True, min=80 

582 """ 

583 if le and ge: 

584 return number 

585 _min = 100 if ge else 60 

586 _max = 100 if le else 140 

587 nb = int(number * self.generator.random.randint(_min, _max) / 100) 

588 if min is not None and nb < min: 

589 nb = min 

590 if max is not None and nb > max: 

591 nb = max 

592 return nb

594 def numerify(self, text: str = "###") -> str: 

595 """Generate a string with each placeholder in ``text`` replaced according 

596 to the following rules: 

597 

598 - Number signs ('#') are replaced with a random digit (0 to 9). 

599 - Percent signs ('%') are replaced with a random non-zero digit (1 to 9). 

600 - Exclamation marks ('!') are replaced with a random digit or an empty string. 

601 - At symbols ('@') are replaced with a random non-zero digit or an empty string. 

602 

603 Under the hood, this method uses :meth:`random_digit() <faker.providers.BaseProvider.random_digit>`, 

604 :meth:`random_digit_not_null() <faker.providers.BaseProvider.random_digit_not_null>`, 

605 :meth:`random_digit_or_empty() <faker.providers.BaseProvider.random_digit_or_empty>`, 

606 and :meth:`random_digit_not_null_or_empty() <faker.providers.BaseProvider.random_digit_not_null_or_empty>` 

607 to generate the random values. 

608 

609 :sample: text='Intel Core i%-%%##K vs AMD Ryzen % %%##X' 

610 :sample: text='!!! !!@ !@! !@@ @!! @!@ @@! @@@' 

611 """ 

612 text = _re_hash.sub(lambda x: str(self.random_digit()), text) 

613 text = _re_perc.sub(lambda x: str(self.random_digit_not_null()), text) 

614 text = _re_excl.sub(lambda x: str(self.random_digit_or_empty()), text) 

615 text = _re_at.sub(lambda x: str(self.random_digit_not_null_or_empty()), text) 

616 return text
